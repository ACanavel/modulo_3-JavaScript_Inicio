El Modelo de Prototipos

JavaScript utiliza un modelo de herencia basado en prototipos, donde los objetos pueden heredar directamente de otros objetos.

Creación de Prototipos


function Persona(nombre) {
    this.nombre = nombre;
}

Persona.prototype.saludar = function() {
    console.log(`Hola, me llamo ${this.nombre}`);
};

const persona1 = new Persona('María');
persona1.saludar(); // Salida: Hola, me llamo María

Herencia Prototípica


function Estudiante(nombre, carrera) {
    Persona.call(this, nombre);
    this.carrera = carrera;
}

Estudiante.prototype = Object.create(Persona.prototype);
Estudiante.prototype.constructor = Estudiante;

Estudiante.prototype.estudiar = function() {
    console.log(`${this.nombre} está estudiando ${this.carrera}`);
};

const estudiante1 = new Estudiante('Juan', 'Medicina');
estudiante1.saludar();   // Salida: Hola, me llamo Juan
estudiante1.estudiar();  // Salida: Juan está estudiando Medicina
